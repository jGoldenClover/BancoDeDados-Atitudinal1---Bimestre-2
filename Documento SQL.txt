CREATE TABLE pedidos (
    name_product VARCHAR(20) PRIMARY KEY,
    item_description TEXT ,
    price_product NUMERIC NOT NULL,
    price_travel NUMERIC NOT NULL,
    payment_form VARCHAR(10) NOT NULL,
    item_request_time TIMESTAMP NOT NULL,
    item_delivery_time TIMESTAMP NOT NULL
  );
